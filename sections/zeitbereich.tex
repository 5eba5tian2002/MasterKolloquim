    \begin{frame}{Tiefpass im Zeitbereich}
        \begin{itemize}
        \item Impulsantwort
        \item Sprungantwort
        \item Faltung
    \end{itemize}
    \begin{itemize}
        \item Beschreibung über die Impulsantwort \(h(t)\)
        \item Ausgangssignal entsteht durch Faltung:
        \[
            y(t) = x(t) * h(t)
        \]
        \item RC-Tiefpass: \(h(t) = \frac{1}{RC} e^{-t/(RC)} u(t)\)
        \item Sprungantwort: \(s(t) = 1 - e^{-t/(RC)}\)
        \item Zeitkonstante \(\tau = RC\) bestimmt Reaktionsgeschwindigkeit
        \item Tiefpass glättet schnelle Änderungen und lässt langsame durch
    \end{itemize}
\end{frame}
